#CUDA7.0導入メモ
#OS:Ubuntu minimal 14.04
#GPU:GTX980Ti

#OSインストール時にsshサーバをインストール

#標準ではcliすら見えないので、sshで作業or真っ暗な中頑張ってgnomeインストールして再起動　
sudo apt-get install gnome

#980Ti対応ドライバのインストール
sudo apt-get install software-properties-common
sudo add-apt-repository ppa:mamarley/nvidia
sudo apt-get update
sudo apt-get install nvidia-352 nvidia-352-dev nvidia-352-uvm nvidia-settings

#CUDA7.0のインストール
wget http://developer.download.nvidia.com/compute/cuda/repos/ubuntu1404/x86_64/cuda-repo-ubuntu1404_7.0-28_amd64.deb
sudo dpkg -i cuda-repo-ubuntu1404_7.0-28_amd64.deb
sudo apt-get install cuda-command-line-tools-7-0 cuda-core-7-0 cuda-cublas-7-0 cuda-cublas-dev-7-0 cuda-cudart-7-0 cuda-cudart-dev-7-0 cuda-cufft-7-0 cuda-cufft-dev-7-0 cuda-curand-7-0 cuda-curand-dev-7-0 cuda-cusolver-7-0 cuda-cusolver-dev-7-0 cuda-cusparse-7-0 cuda-cusparse-dev-7-0 cuda-documentation-7-0 cuda-driver-dev-7-0 cuda-license-7-0 cuda-misc-headers-7-0 cuda-npp-7-0 cuda-npp-dev-7-0 cuda-nvrtc-7-0 cuda-nvrtc-dev-7-0 cuda-samples-7-0 cuda-toolkit-7-0 cuda-visual-tools-7-0 freeglut3 freeglut3-dev nvidia-modprobe
